    <p>Loading…</p>

  </div>


  <script>

    fetch('medications.json')

      .then(res => {

        if (!res.ok) throw new Error('Network response was not ok');

        return res.json();

      })

      .then(data => {

        const tree = document.getElementById('tree');

        tree.innerHTML = '';

        const ul = document.createElement('ul');

        ul.className = 'tree';


        data.forEach(r => {

          const li = document.createElement('li');

          // Medication name

          const lbl = document.createElement('span');

          lbl.textContent = r.Medication || 'Unnamed';

          lbl.className = 'label';


          // Detail list

          const detailUl = document.createElement('ul');

          detailUl.className = 'children';

          ['Type','Strength','Quantity','Price','Standard Dosing','State']

            .forEach(k => {

              if (r[k]) {

                const dLi = document.createElement('li');

                dLi.textContent = `${k}: ${r[k]}`;

                detailUl.appendChild(dLi);

              }

            });


          // Pharmacy availability

          const pharmUl = document.createElement('ul');

          pharmUl.className = 'children';

          ['Hallandale','1st Choice','Rite Away','REVIVE','NuBioage','ScriptWorks']

            .forEach(ph => {

              if (r[ph] && r[ph].toString().toLowerCase() === 'true') {

                const pLi = document.createElement('li');

                pLi.textContent = `Available from ${ph}`;

                pharmUl.appendChild(pLi);

              }

            });


          // Toggle details & pharmacy on click

          lbl.addEventListener('click', () => {

            detailUl.classList.toggle('open');

            pharmUl.classList.toggle('open');

          });


          li.appendChild(lbl);

          li.appendChild(detailUl);

          li.appendChild(pharmUl);

          ul.appendChild(li);

        });


        tree.appendChild(ul);

      })

      .catch(err => {

        console.error(err);

        document.getElementById('tree').textContent = 'Error loading data.';

      });

  </script>
